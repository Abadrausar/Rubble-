PACKAGE

package ini
    import "dctech/nca7/commands/ini"

    NCA v7 INI Commands.

FUNCTIONS

func CommandIni_Create(state *nca7.State, params []*nca7.Value)
    Create a new (empty) ini file.

	ini:create

    Returns the ini file

func CommandIni_Format(state *nca7.State, params []*nca7.Value)
    Format an ini file for writing to disk.

	ini:format file

    Returns the string representation of the ini file or an error message.
    May set the Error flag.

func CommandIni_GetValue(state *nca7.State, params []*nca7.Value)
    Gets a specific value from an ini file.

	ini:getvalue file sectionname keyname

    Returns the value or an error message. May set the Error flag.

func CommandIni_Parse(state *nca7.State, params []*nca7.Value)
    Parse an ini file.

	ini:parse filecontents

    Returns the ini file or an error message. May set the Error flag.

func CommandIni_SetValue(state *nca7.State, params []*nca7.Value)
    Sets a specific value in an ini file.

	ini:setvalue file sectionname keyname value

    Returns unchanged or an error message. May set the Error flag.

func Setup(state *nca7.State)
    Adds the ini commands to the state. The ini commands are:

	ini:create
	ini:parse
	ini:format
	ini:getvalue
	ini:setvalue
