PACKAGE

package cmp
    import "dctech/nca6/cmp"

    NCA v6 Integer Comparison Commands.

FUNCTIONS

func CommandEq(state *nca6.State, params []*nca6.Value)
    Integer equal.

	eq a b

    Returns 0 or -1

func CommandGt(state *nca6.State, params []*nca6.Value)
    Integer greater than.

	gt a b

    Returns 0 or -1

func CommandLt(state *nca6.State, params []*nca6.Value)
    Integer less than.

	lt a b

    Returns 0 or -1

func Setup(state *nca6.State)
    Adds the integer comparison commands to the state. The integer
    comparison commands are:

	gt
	lt
	eq
