PACKAGE

package bit
    import "dctech/nca6/bit"

    NCA v6 Bitwise Commands.

FUNCTIONS

func CommandAnd(state *nca6.State, params []*nca6.Value)
    Ands two values.

	and a b

    Returns a & b

func CommandLShift(state *nca6.State, params []*nca6.Value)
    Shifts the value left. Count values of 0 or less are given the value 1.

	lshift a [count]

    Returns the param shifted "count" places

func CommandNot(state *nca6.State, params []*nca6.Value)
    Inverts a value.

	not a

    Returns a ^ -1

func CommandOr(state *nca6.State, params []*nca6.Value)
    Ors two values.

	or a b

    Returns a | b

func CommandRShift(state *nca6.State, params []*nca6.Value)
    Shifts the value right. Count values of 0 or less are given the value 1.

	rshift a [count]

    Returns the param shifted "count" places

func CommandXor(state *nca6.State, params []*nca6.Value)
    Xors two values.

	xor a b

    Returns a ^ b

func Setup(state *nca6.State)
    Adds the bitwise commands to the state. The bitwise commands are:

	lshift
	rshift
	and
	or
	xor
	not
