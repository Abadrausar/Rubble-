PACKAGE

package bit
    import "dctech/nca7/commands/bit"

    NCA v7 Bitwise Commands.

FUNCTIONS

func CommandBit_And(state *nca7.State, params []*nca7.Value)
    Ands two values.

	bit:and a b

    Returns a & b

func CommandBit_LShift(state *nca7.State, params []*nca7.Value)
    Shifts the value left. Count values of 0 or less are given the value 1.

	bit:lshift a [count]

    Returns the param shifted "count" places

func CommandBit_Not(state *nca7.State, params []*nca7.Value)
    Inverts a value.

	bit:not a

    Returns a ^ -1

func CommandBit_Or(state *nca7.State, params []*nca7.Value)
    Ors two values.

	bit:or a b

    Returns a | b

func CommandBit_RShift(state *nca7.State, params []*nca7.Value)
    Shifts the value right. Count values of 0 or less are given the value 1.

	bit:rshift a [count]

    Returns the param shifted "count" places

func CommandBit_Xor(state *nca7.State, params []*nca7.Value)
    Xors two values.

	bit:xor a b

    Returns a ^ b

func Setup(state *nca7.State)
    Adds the bitwise commands to the state. The bitwise commands are:

	bit:lshift
	bit:rshift
	bit:and
	bit:or
	bit:xor
	bit:not
