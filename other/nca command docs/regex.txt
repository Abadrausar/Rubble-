PACKAGE

package regex
    import "dctech/nca7/commands/regex"

    NCA v7 Regular Expression Commands.

FUNCTIONS

func CommandRegEx_Replace(state *nca7.State, params []*nca7.Value)
    Runs a regular expression search and replace.

	regex:replace regex input replace

    Returns input with all strings matching regex replaced with replace.

func Setup(state *nca7.State)
    Adds the regex commands to the state. The regex commands are:

	regex:replace
