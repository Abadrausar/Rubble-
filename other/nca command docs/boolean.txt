PACKAGE

package boolean
    import "dctech/nca7/commands/boolean"

    NCA v7 Boolean Commands.

FUNCTIONS

func CommandBool_And(state *nca7.State, params []*nca7.Value)
    Ands two values.

	bool:and a b

    Returns a && b

func CommandBool_Not(state *nca7.State, params []*nca7.Value)
    Inverts a value.

	bool:not a

    Returns !a

func CommandBool_Or(state *nca7.State, params []*nca7.Value)
    Ors two values.

	bool:or a b

    Returns a || b

func Setup(state *nca7.State)
    Adds the boolean commands to the state. The boolean commands are:

	int:and
	int:or
	int:not
