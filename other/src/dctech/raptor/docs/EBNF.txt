/* 
 EBNF for Raptor
 
 I am not very good at BNF, so this probably has "issues",
 in any case the actual implementation is the correct way.
*/

newline = "\n" | "\r" .

/* Yes, raptor treats commas as whitespace, if you want to use commas as delimiters go ahead */
whitespace = " " | "\t" | "," .

reservedchar = "[" | "]" | "{" | "}" | "<" | ">" | "(" | ")" .

letter = /* any character except newline, whitespace, or a reserved char */ .

comment = "#" { letter | whitespace | reservedchar } newline .

rawstring = letter { letter } .

dqstring = "\"" { letter | whitespace | reservedchar | newline } "\"" .

string = rawstring | dqstring .

program = { value } .

value = string | command | dereference | block | objlit .

command = "(" value { value } ")" .
dereference = "[" value { value } "]" .
block = "{" { value } "}" .
objlit = "<" value { value "=" value } ">" | "<" value { value } ">" .
