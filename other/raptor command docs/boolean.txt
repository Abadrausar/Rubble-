PACKAGE

package boolean
    import "dctech/raptor/commands/boolean"

    Raptor Boolean Commands.

FUNCTIONS

func CommandBool_And(script *raptor.Script, params []*raptor.Value)
    Ands two values.

	bool:and a b

    Returns a && b

func CommandBool_Not(script *raptor.Script, params []*raptor.Value)
    Inverts a value.

	bool:not a

    Returns !a

func CommandBool_Or(script *raptor.Script, params []*raptor.Value)
    Ors two values.

	bool:or a b

    Returns a || b

func Setup(state *raptor.State)
    Adds the boolean commands to the state. The boolean commands are:

	bool:and
	bool:or
	bool:not
